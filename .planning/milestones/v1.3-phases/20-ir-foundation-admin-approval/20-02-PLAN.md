async function checkAdminAuth() {
  const session = await auth.api.getSession({ headers: await headers() })
  if (!session) throw new Error("Unauthorized")
  const [dbUser] = await db.select({ role: user.role }).from(user).where(eq(user.id, session.user.id)).limit(1)
  if (!dbUser || dbUser.role !== "admin") throw new Error("Unauthorized")
  return session
}

**getTeamIrSlots(teamId: number, leagueId: number):**
- Query irRequests joined with riders (for riderName)
- Filter: teamId = teamId AND leagueId = leagueId
- Order: submittedAt DESC
- Return: array of IrSlot objects

**getActiveRosterCount(teamId: number, leagueId: number):**
- Query 1: COUNT(*) from draftPicks where teamId = teamId AND leagueId = leagueId
- Query 2: COUNT(*) from irRequests where teamId = teamId AND leagueId = leagueId AND status = "approved"
- Return: totalPicks - approvedIr (as number)
- This value tells the transfer submission guard how many active slots the team actually uses

**getPendingIrRequests():**
- Query irRequests joined with leagues (leagueName), teams (teamName), riders (riderName)
- Filter: status = "pending"
- Order: submittedAt ASC
- Return: array of PendingIrRequest objects

**src/app/(main)/leagues/[leagueId]/ir/actions.ts:**

Add `"use server"` at top.

`submitIrRequest(data: { leagueId: number; riderId: number; reason?: string })`:

1. Auth via `getAuthenticatedUser()` — return error if not authed
2. Check league membership via `checkLeagueMembership(userId, leagueId)` — return error if not member
3. Verify rider is on this team: query draftPicks where teamId = team.id AND leagueId AND riderId = data.riderId — return error "Rider is not on your team" if not found
4. Count occupied IR slots: query irRequests where teamId = team.id AND leagueId = data.leagueId AND status IN ("pending", "approved") — return error "IR slots full (max 2)" if count >= 2
5. Check no duplicate: query irRequests where teamId = team.id AND riderId = data.riderId AND leagueId = data.leagueId AND status IN ("pending", "approved") — return error "This rider already has an active IR request" if found
6. Insert irRequests row with status "pending"
7. `revalidatePath(\`/leagues/${data.leagueId}/ir\`)` and `revalidatePath("/admin/ir")`
8. Return `{ success: true }`

5. revalidatePath("/admin/ir")
6. revalidatePath(`/leagues/${request.leagueId}/ir`)
