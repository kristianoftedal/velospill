```sql
-- Create order_status enum
CREATE TYPE order_status AS ENUM ('pending', 'active', 'rejected', 'countered');

-- Create orders table
CREATE TABLE IF NOT EXISTS orders (
  id SERIAL PRIMARY KEY,
  "leagueId" INTEGER NOT NULL REFERENCES leagues(id) ON DELETE CASCADE,
  "teamId" INTEGER NOT NULL REFERENCES teams(id),
  "raceId" INTEGER NOT NULL REFERENCES races(id),
  "orderTypeId" INTEGER NOT NULL REFERENCES "orderTypes"(id),
  status order_status NOT NULL DEFAULT 'pending',
  "targetRiderId" INTEGER REFERENCES riders(id),
  "targetTeamId" INTEGER REFERENCES teams(id),
  "targetProTeam" TEXT,
  "targetCountry" TEXT,
  "orderConfig" JSONB,
  "bonusPoints" INTEGER,
  "adminNote" TEXT,
  "submittedAt" TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  "resolvedAt" TIMESTAMPTZ,
  "resolvedBy" TEXT REFERENCES "user"(id)
);

-- Create indexes
CREATE INDEX IF NOT EXISTS orders_league_idx ON orders ("leagueId");
CREATE INDEX IF NOT EXISTS orders_team_idx ON orders ("teamId");
CREATE INDEX IF NOT EXISTS orders_race_idx ON orders ("raceId");
CREATE UNIQUE INDEX IF NOT EXISTS orders_team_race_unique ON orders ("teamId", "raceId");
```

Use `Pool` from `@neondatabase/serverless` with `pool.connect()` then `client.query(sql)` then `client.release()` — exactly as done in Phase 06 migrations (STATE.md decision #06-01: Pool.connect() for DDL). Read the DATABASE_URL from process.env.

After migration, delete any temporary migration script file. Do NOT leave migration scripts checked in.
