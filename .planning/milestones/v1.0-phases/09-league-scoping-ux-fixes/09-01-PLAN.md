   ```typescript
     await tx
       .update(draftSessions)
       .set({
         currentPickIndex: nextPickIndex,
         currentTeamId: nextTeamId,
         currentGender: nextGender,
         timerExpiresAt: nextTimerExpiresAt,
         status: nextStatus as "men" | "women" | "complete" | "pending" | "paused",

     }
   })
   ```
