Create src/app/(main)/riders/[riderId]/page.tsx as a Next.js App Router server component.

Pattern follows the established project convention (see standings/page.tsx):
- `interface PageProps { params: Promise<{ riderId: string }> }`
- Await params, parseInt riderId — call `notFound()` if NaN
- Call `getRiderSeasonProfile(riderId)` from `@/lib/rider-queries`
- Call `notFound()` if the function returns null
- Pass data to `<RiderProfileClient>` client component

Full implementation:
```tsx
import { notFound } from "next/navigation"
import Link from "next/link"
import { getRiderSeasonProfile } from "@/lib/rider-queries"
import RiderProfileClient from "./rider-profile-client"

interface PageProps {
  params: Promise<{ riderId: string }>
}

export default async function RiderProfilePage({ params }: PageProps) {
  const { riderId: riderIdStr } = await params
  const riderId = parseInt(riderIdStr, 10)

  if (isNaN(riderId)) {
    notFound()
  }

  const profile = await getRiderSeasonProfile(riderId)

  if (!profile) {
    notFound()
  }

  return (
    <div className="container mx-auto max-w-4xl px-4 py-8 space-y-6">
      <nav className="text-sm text-gray-500">
        <Link href="/riders" className="hover:text-gray-700 hover:underline">
          Riders
        </Link>
        <span className="mx-2">&rsaquo;</span>
        <span className="text-gray-900">{profile.rider.name}</span>
      </nav>
      <RiderProfileClient profile={profile} />
    </div>
  )
}
```
